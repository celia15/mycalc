
LineList ::= Line +
Line ::= Expression Eol
Expression ::= Term ( ( Add | Sub )  Term ) *
Term ::= PrimaryExpression  ( ( Mul | Div ) PrimaryExpression ) *
PrimaryExpression ::= Sub ? (DoubleLiteral | ( LParen Expression RParen ))

Add ::= '+'
Sub ::= '-'
Minus ::= '-'
Mul ::= '*'
Div ::= '/'
DoubleLiteral ::= [0-9]+ ('.' [0-9]+)?
Eol ::= '\n'
