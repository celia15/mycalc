
LineList ::= Line +
Line ::= Expression Eol
Expression ::= Term ( ( Add | Sub )  Term ) *
Term ::= PrimaryExpression  ( ( Mul | Div ) PrimaryExpression ) *
PrimaryExpression ::=
    Sushi
    | Variable Assign Expression
    | Variable
    | Sub ? DoubleLiteral
    | ( LParen Expression RParen )

Add ::= '+'
Sub ::= '-'
Minus ::= '-'
Mul ::= '*'
Div ::= '/'
DoubleLiteral ::= [0-9]+ ('.' [0-9]+)?
Sushi ::= üç£
Variable ::= [a-zA-Z]+
Assign ::= '='
Eol ::= '\n'
